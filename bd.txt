CREATE TABLE IF NOT EXISTS sga.usuarios
(
    id serial NOT NULL PRIMARY KEY,
    nome varchar(100),
    login varchar(50),
    senha text,
    status boolean,
    tipo varchar(10),
    guiche varchar(10),
    fk_servico bigint,
    CONSTRAINT fk_user_service FOREIGN KEY (fk_servico)
        REFERENCES sga.servicos (id) MATCH SIMPLE
        ON UPDATE NO ACTION
        ON DELETE NO ACTION
);

CREATE TABLE IF NOT EXISTS sga.servicos
(
    id serial NOT NULL PRIMARY KEY,
    description varchar(30),
    status boolean,
    departamento varchar(250),
    pd_senha varchar(5),
    CONSTRAINT pd_senha_unique UNIQUE (pd_senha)
);

CREATE TABLE IF NOT EXISTS sga.configs
(   
    id serial NOT NULL PRIMARY KEY,
    son text,
    mensagem text,
    conteudo text,
    tipo_conteudo varchar(10)
);
